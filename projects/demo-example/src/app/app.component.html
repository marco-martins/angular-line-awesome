Angular bind sintaxe:

iconName: IconProp = ['las', 'hippo'];
iconSize: string = '2x';

<la-icon [icon]="iconName" [size]="iconSize"></la-icon>

Angular bind sintaxe:

transform: Transform = {size: 15, x: 5, y: -5, rotate: 90, flipX: true, flipY: true};

<la-icon icon="las hippo" [transform]="transform"></la-icon>

Stacked icons:

heartIcon: IconProp = {prefix: 'las', iconName: 'heart'};
angularIcon: IconProp = {prefix: 'lab', iconName: 'angular'}; 

<la-stack size="2x">
  <la-icon [icon]="heartIcon" stackItemSize="2x"></la-icon>
  <la-icon [icon]="angularIcon" stackItemSize="1x" [inverse]="true"></la-icon>
</la-stack>
